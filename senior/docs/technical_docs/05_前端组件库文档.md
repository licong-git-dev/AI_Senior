## 概述

### 项目简介
养老智能体前端系统是基于React 18 + TypeScript构建的现代化SPA应用，专为老年人群设计的智能养老服务平台。

### 核心特性
- **适老化设计**: 大字号、高对比度、简化操作
- **多端适配**: 响应式设计，支持手机、平板、电脑
- **实时交互**: WebSocket实时数据更新
- **无障碍支持**: 完整的可访问性支持
- **PWA支持**: 离线使用、安装到桌面

## 适老化组件

### 适老化设计原则
```yaml
大字号: 默认16px，最小14px，可调节至24px
高对比: 对比度≥4.5:1，支持深色模式
大按钮: 最小44x44px触摸目标
简导航: 面包屑+返回按钮+语音导航
防误触: 二次确认+撤销机制
```

### 大字号组件
```tsx
// ResponsiveText.tsx
import { useEffect, useState } from 'react';

interface ResponsiveTextProps {
  children: React.ReactNode;
  baseSize?: number;
  size?: 'small' | 'medium' | 'large' | 'xlarge';
  className?: string;
}

export const ResponsiveText: React.FC<ResponsiveTextProps> = ({
  children,
  baseSize = 16,
  size = 'medium',
  className = ''
}) => {
  const [fontSize, setFontSize] = useState(baseSize);
  
  const sizeMultiplier = {
    small: 0.875,
    medium: 1,
    large: 1.25,
    xlarge: 1.5
  }[size];
  
  useEffect(() => {
    const savedSize = localStorage.getItem('font-size');
    if (savedSize) {
      setFontSize(parseInt(savedSize));
    }
  }, []);
  
  return (
    <span 
      className={`${className}`}
      style={{ fontSize: `${fontSize * sizeMultiplier}px` }}
    >
      {children}
    </span>
  );
};
```

### 语音输入组件
```tsx
// VoiceInput.tsx
import { useState, useRef } from 'react';
import { Mic, MicOff } from 'lucide-react';

interface VoiceInputProps {
  onTranscript: (text: string) => void;
  placeholder?: string;
  className?: string;
}

export const VoiceInput: React.FC<VoiceInputProps> = ({
  onTranscript,
  placeholder = '按住说话或点击录音',
  className = ''
}) => {
  const [isRecording, setIsRecording] = useState(false);
  const [transcript, setTranscript] = useState('');
  const recognitionRef = useRef<SpeechRecognition | null>(null);
  
  const startRecording = () => {
    const SpeechRecognition = window.SpeechRecognition || window.webkitSpeechRecognition;
    if (!SpeechRecognition) {
      alert('您的浏览器不支持语音识别');
      return;
    }
    
    const recognition = new SpeechRecognition();
    recognition.continuous = false;
    recognition.interimResults = false;
    recognition.lang = 'zh-CN';
    
    recognition.onresult = (event) => {
      const text = event.results[0][0].transcript;
      setTranscript(text);
      onTranscript(text);
      setIsRecording(false);
    };
    
    recognition.onerror = () => {
      setIsRecording(false);
    };
    
    recognition.onend = () => {
      setIsRecording(false);
    };
    
    recognitionRef.current = recognition;
    recognition.start();
    setIsRecording(true);
  };
  
  const stopRecording = () => {
    recognitionRef.current?.stop();
    setIsRecording(false);
  };
  
  return (
    <div className={`flex items-center space-x-2 ${className}`}>
      <button
        onClick={isRecording ? stopRecording : startRecording}
        className={`
          p-3 rounded-full border-2 transition-all duration-200
          ${isRecording 
            ? 'bg-red-500 text-white border-red-500 animate-pulse' 
            : 'bg-blue-500 text-white border-blue-500 hover:bg-blue-600'
          }
        `}
        aria-label={isRecording ? '停止录音' : '开始录音'}
      >
        {isRecording ? <MicOff size={24} /> : <Mic size={24} />}
      </button>
      
      <div className="flex-1">
        <input
          type="text"
          value={transcript}
          onChange={(e) => setTranscript(e.target.value)}
          placeholder={placeholder}
          className="w-full p-3 border-2 border-gray-300 rounded-lg text-lg"
        />
      </div>
    </div>
  );
};
```

```tsx
import { Slot } from '@radix-ui/react-slot';

import { cn } from '@/lib/utils';

const buttonVariants = cva(
  'inline-flex items-center justify-center whitespace-nowrap rounded-md text-sm font-medium transition-colors focus-visible:outline-none focus-visible:ring-2 focus-visible:ring-ring focus-visible:ring-offset-2 disabled:pointer-events-none disabled:opacity-50',

        secondary: 'bg-secondary text-secondary-foreground hover:bg-secondary/80',
        ghost: 'hover:bg-accent hover:text-accent-foreground',

        sm: 'h-9 rounded-md px-3',
        lg: 'h-11 rounded-md px-8',
        xl: 'h-14 rounded-lg px-12 text-lg',
        icon: 'h-10 w-10',
      },
    },
    defaultVariants: {
      variant: 'default',
      size: 'default',
    },
  }
);

export interface ButtonProps
  extends React.ButtonHTMLAttributes<HTMLButtonElement>,
    VariantProps<typeof buttonVariants> {
  asChild?: boolean;
}

const Button = React.forwardRef<HTMLButtonElement, ButtonProps>(
  ({ className, variant, size, asChild = false, ...props }, ref) => {
    const Comp = asChild ? Slot : 'button';
    return (
      <Comp
        className={cn(buttonVariants({ variant, size, className }))}
        ref={ref}
        {...props}
      />
    );
  }
);
Button.displayName = 'Button';

export { Button, buttonVariants };
```

```tsx
// ElderlyDashboard.tsx
import { useEffect, useState } from 'react';

import { DeviceStatus } from '@/components/DeviceStatus';
import { EmergencyPanel } from '@/components/EmergencyPanel';
import { ResponsiveText } from '@/components/ResponsiveText';

export default function ElderlyDashboard() {
  const [user, setUser] = useState<any>(null);
  const [healthMetrics, setHealthMetrics] = useState<any[]>([]);
  const [devices, setDevices] = useState<any[]>([]);
  const [loading, setLoading] = useState(true);
  
  useEffect(() => {
    loadDashboardData();
  }, []);
  
  const loadDashboardData = async () => {
    try {
      // 加载用户信息
      const { data: userData } = await supabase
        .from('profiles')
        .select('*')
        .single();
      setUser(userData);
      
      // 加载最新健康数据
      const { data: healthData } = await supabase
        .from('health_data')
        .select('*')
        .order('measurement_time', { ascending: false })
        .limit(4);
      setHealthMetrics(healthData || []);
      
      // 加载设备状态
      const { data: deviceData } = await supabase
        .from('devices')
        .select('*')
        .eq('status', 1);
      setDevices(deviceData || []);
      
    } catch (error) {
      console.error('加载仪表板数据失败:', error);
    } finally {
      setLoading(false);
    }
  };
  
  if (loading) {
    return (
      <div className="min-h-screen flex items-center justify-center">
        <div className="text-center">
          <div className="animate-spin rounded-full h-16 w-16 border-b-2 border-blue-600 mx-auto"></div>

              year: 'numeric', 
              month: 'long', 
              day: 'numeric',
              weekday: 'long'
            })}

        {/* 紧急呼叫面板 */}

        {/* 健康指标 */}

          <ResponsiveText size="large" className="text-xl font-bold text-gray-800 mb-4">
            健康状况
          </ResponsiveText>
          <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-4 gap-6">
            {healthMetrics.map((metric, index) => (
              <HealthMetricCard
                key={index}
                title={metric.data_type}
                value={metric.data_value}
                unit={metric.unit}
                normalRange={getNormalRange(metric.data_type)}
                status={getHealthStatus(metric)}
                trend={getTrend(metric)}

            ))}
```

```tsx
        .from('devices')
        .insert([deviceData])
        .select()
        .single();
      
      if (insertError) throw insertError;
      
      setDevices(prev => [...prev, data]);
      return data;
    } catch (err) {
      setError(err instanceof Error ? err.message : '连接设备失败');
      throw err;
    }
  };
  
  const disconnectDevice = async (deviceId: string) => {
    try {
      const { error: deleteError } = await supabase
        .from('devices')
        .delete()
        .eq('id', deviceId);
      
      if (deleteError) throw deleteError;

    } catch (err) {
      setError(err instanceof Error ? err.message : '断开设备失败');
      throw err;
    }
  };

    loading,
```

```tsx
import { useEffect, useRef } from 'react';
import { supabase } from '@/lib/supabase';

interface UseRealtimeProps {
  channel: string;
  event: 'INSERT' | 'UPDATE' | 'DELETE' | '*';
  table: string;
  filter?: string;
  onMessage: (payload: any) => void;
}

export const useRealtime = ({
  channel,
  event,
  table,
  filter,
  onMessage
}: UseRealtimeProps) => {
  const subscriptionRef = useRef<any>(null);
  
  useEffect(() => {
    const subscription = supabase
      .channel(channel)
      .on(
        'postgres_changes',
        {
          event,
          schema: 'public',
          table,
          filter
        },
        onMessage
      )
      .subscribe();
    
    subscriptionRef.current = subscription;
    
    return () => {
      if (subscriptionRef.current) {
        supabase.removeChannel(subscriptionRef.current);
      }
    };
  }, [channel, event, table, filter, onMessage]);
  
  return subscriptionRef.current;
};
```

```tsx
// contexts/HealthContext.tsx
import { createContext, useContext, useReducer, useEffect } from 'react';
import { supabase } from '@/lib/supabase';

interface HealthData {
  id: string;
  data_type: string;
  data_value: number;
  measurement_time: string;
  abnormal_flag: number;
}

interface HealthState {
  healthData: HealthData[];
  loading: boolean;
  error: string | null;
  lastUpdate: string | null;
}

type HealthAction = 
  | { type: 'SET_LOADING'; payload: boolean }
  | { type: 'SET_HEALTH_DATA'; payload: HealthData[] }
  | { type: 'ADD_HEALTH_DATA'; payload: HealthData }
  | { type: 'UPDATE_HEALTH_DATA'; payload: HealthData }
  | { type: 'SET_ERROR'; payload: string }
  | { type: 'CLEAR_ERROR' };

const initialState: HealthState = {
  healthData: [],
  loading: false,
  error: null,
  lastUpdate: null
};

const healthReducer = (state: HealthState, action: HealthAction): HealthState => {
  switch (action.type) {
    case 'SET_LOADING':
      return { ...state, loading: action.payload };
    
    case 'SET_HEALTH_DATA':
      return { 
        ...state, 
        healthData: action.payload,
        loading: false,
        lastUpdate: new Date().toISOString()
      };
    
    case 'ADD_HEALTH_DATA':
      return {
        ...state,
        healthData: [action.payload, ...state.healthData],
        lastUpdate: new Date().toISOString()
      };
    
    case 'UPDATE_HEALTH_DATA':
      return {
        ...state,
        healthData: state.healthData.map(item =>
          item.id === action.payload.id ? action.payload : item
        ),
        lastUpdate: new Date().toISOString()
      };
    
    case 'SET_ERROR':
      return { ...state, error: action.payload, loading: false };
    
    case 'CLEAR_ERROR':
      return { ...state, error: null };
    
    default:
      return state;
  }
};

interface HealthContextType {
  state: HealthState;
  fetchHealthData: (userId: string) => Promise<void>;
  addHealthData: (data: Omit<HealthData, 'id'>) => Promise<void>;
  clearError: () => void;
}

const HealthContext = createContext<HealthContextType | undefined>(undefined);

export const HealthProvider: React.FC<{ children: React.ReactNode }> = ({ children }) => {
  const [state, dispatch] = useReducer(healthReducer, initialState);
  
  useEffect(() => {
    // 订阅实时健康数据更新
    const subscription = supabase
      .channel('health_data_realtime')
      .on(
        'postgres_changes',
        {
          event: 'INSERT',
          schema: 'public',
          table: 'health_data'
        },
        (payload) => {
          dispatch({ type: 'ADD_HEALTH_DATA', payload: payload.new as HealthData });
        }
      )
      .subscribe();
    
    return () => {
      subscription.unsubscribe();
    };
  }, []);
  
  const fetchHealthData = async (userId: string) => {
    dispatch({ type: 'SET_LOADING', payload: true });
    dispatch({ type: 'CLEAR_ERROR' });
    
    try {
      const { data, error } = await supabase
        .from('health_data')
        .select('*')
        .eq('user_id', userId)
        .order('measurement_time', { ascending: false })
        .limit(100);
      
      if (error) throw error;
      
      dispatch({ type: 'SET_HEALTH_DATA', payload: data || [] });
    } catch (error) {
      dispatch({ 
        type: 'SET_ERROR', 
        payload: error instanceof Error ? error.message : '获取健康数据失败' 
      });
    }
  };
  
  const addHealthData = async (data: Omit<HealthData, 'id'>) => {
    try {
      const { data: result, error } = await supabase
        .from('health_data')
        .insert([data])
        .select()
        .single();
      
      if (error) throw error;
      
      dispatch({ type: 'ADD_HEALTH_DATA', payload: result });
    } catch (error) {
      dispatch({ 
        type: 'SET_ERROR', 
        payload: error instanceof Error ? error.message : '添加健康数据失败' 
      });
      throw error;
    }
  };
  
  const clearError = () => {
    dispatch({ type: 'CLEAR_ERROR' });
  };
  
  return (
    <HealthContext.Provider value={{
      state,
      fetchHealthData,
      addHealthData,
      clearError
    }}>
      {children}
    </HealthContext.Provider>
  );
};

export const useHealth = () => {
  const context = useContext(HealthContext);
  if (context === undefined) {
    throw new Error('useHealth must be used within a HealthProvider');
  }
  return context;
};
```
